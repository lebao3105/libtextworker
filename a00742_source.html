21 def CreateMenu(parent, items: list[tuple[5]]) -> wx.Menu:
22  """
23  Create a new wx.Menu with a list of commands.
24  Menu items use the following format:
25  ```
26  (id, label, helptext, handler, kind) # Stay in a tuple like this
27  ```
28  * id: Menu item id (wx.ID_* - ID_ANY is the default)
29  * label: Menu item label
30  * helptext: Text to be shown in the status bar (on the left)
31  * handler (Callable): Callback
32  * kind: Type of the menu item (wx.ITEM_*) (wx.ITEM_NORMAL by default)
33 
34  If you want to append a separator, make all items in the tuple None.
35  Sub-menus are not supported.
36  Returns the generated menu.
37  """
38  target_menu = wx.Menu()
39  for id, label, helptext, handler, kind in items:
40  if id == label == helptext == handler == kind == None:
41  target_menu.AppendSeparator()
42  else:
43  if id == None:
44  id = wx.ID_ANY
45  if kind == None:
46  kind = wx.ITEM_NORMAL
47  if label == None:
48  label = ""
49  if helptext == None:
50  helptext = ""
51  item = target_menu.Append(id, label, helptext, kind)
52  parent.Bind(wx.EVT_MENU, handler, item)
53  return target_menu

137 def localizePy(path: str, importText: str | None = None, ignoreDoneWork: bool = True) -> ModuleType:
138  """
139  Localizes wxFormBuilder-generated Python code.
140  @param path : The path to the Python code
141  @param importText : The from .. import statement that will be added to the file (like importing gettext)
142  @param ignoreDoneWork: Ignore translated file
143  @returns result (ModuleType): Imported @path
144  """
145 
146  assert os.path.isfile(path)
147  content = open(path, "r").read()
148 
149  if not ignoreDoneWork:
150  assert content.startswith(path) == False, \
151  "Already translated. To avoid the file being broken, this cannot continue.\n" \
152  "Replace the file with newly generated code from wxFormBuilder and try again."
153 
154  if not importText.endswith("\n"): importText += "\n"
155  func = importText.split(" ")[-1] # Get the translate function (e.g gettext in from gettext import gettext)
156  content = importText + content # Add import statement
157 
158  # Made with the help of, yeah, AI. (cuz I'm bad at regex)
159  # I have modified it myself for the most perfect pattern.
160  # All strings in wxFormBuilder generated Python code are unicode type:
161  # u"<content>"
162  # Pattern with start with u. Then 4 regex groups:
163  # One for the first double quote (")
164  # One for the string content (. for every character except \n, + for >=1 match, ? to catch as much as posible)
165  # The matching quote (")
166  # The final look for space (just one space) OR a comma (,) but NOT include it to the catch result.
167  # One draw back is that numberic only strings are included, but can be skipped below.
168  pattern = r'u(")(.+?)(")(?=[\s,])'
169  matches = re.findall(pattern, content)
170 
171  for match in matches:
172  # The match result will be a list of
173  # (quote, string, matching quote) tuples.
174  # Skip numberic/float strings.
175  try: int(match[1]); float(match[1])
176  except: pass
177  else: continue
178  localized = f'{func}(u"{match[1]}")'
179  content = content.replace(f'u"{match[1]}"', localized)
180 
181  open(path, "w").write(content)
182  name = os.path.basename(path)
183  dirpath = path.removesuffix(name)
184 
185  import importlib.util
186  spec = importlib.util.spec_from_file_location(name, path)
187  assert spec != None
188  result = importlib.util.module_from_spec(spec)
189  spec.loader.exec_module(result)
190  return result
