5def CreateMenu(parent, items: list) -> wx.Menu:
6 """
7 Create a new wx.Menu with a list of commands.
8 Menu items use the following format:
9 ```
10 [ # Must be a list
11 (id, label, helptext, handler, None),
12 ...
13 (id, label, helptext, handler, wx.ITEM_CHECK), # Add check item
14 (id, label, helptext, handler, wx.ITEM_RADIO) # Add radio item
15 (None, None, None, None, None) # Add separator
16 ]
17 ```
18 Sub-menus are not supported.
19 Don't forget to set the parent of the menubar first! (SetParent function)
20 Returns the generated menu.
21 """
22 target_menu = wx.Menu()
23
24 for id, label, helptext, handler, kind in items:
25 if id == label == helptext == handler == kind == None:
26 target_menu.AppendSeparator()
27 else:
28 if id == None:
29 id = wx.ID_ANY
30 if kind == None:
31 kind = wx.ITEM_NORMAL
32 if label == None:
33 label = ""
34 if helptext == None:
35 helptext = ""
36 item = target_menu.Append(id, label, helptext, kind)
37 parent.Bind(wx.EVT_MENU, handler, item)
38 return target_menu
