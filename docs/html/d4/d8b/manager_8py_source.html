195 def configure(self, widget: typing.Any):
196 """
197 Configure a widget with pre-defined settings.
198 This function also ables to rerun itself under a threading thread.
199 @param widget : Widget to configure
200 @see setcolorfunc
201 @see setfontcfunc
202 """
203 if not widget:
204 print("Widget died, skip configuring.")
205 return

211 fn = self.setfontfn[item]["fn"]
